function getIndexToIns(arr, num) {
  if (!arr.length) {
    return 0;
  }

  const sortedArr = arr.sort((a, b) => a - b);
 
  const index = sortedArr.findIndex(element1 => element1 >= num);
  return index === -1? sortedArr.length : index;
}

console.log(getIndexToIns([3, 10, 5], 11));
