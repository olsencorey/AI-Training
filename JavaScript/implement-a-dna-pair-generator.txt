function pairElement(str) {
  const pairs = {
    A: "T",
    T: "A",
    C: "G",
    G: "C"
  }
  const result = [];
  for (let base of str) {
    result.push([base, pairs[base]]);
  }

  return result;
}

console.log(pairElement("CTCTA"));
