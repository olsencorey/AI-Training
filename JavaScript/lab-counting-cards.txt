let count = 0;

function cc(card) {
  if (typeof card === 'number' && card > 1 && card <= 10) {
    if (card < 7) {
      count += 1;
    } else if (card == 10) {
      count -=1;
    }
  } else if (typeof card === 'string') {
    count -= 1;
  }
  
  if (count <= 0) {
    return count + " Hold";
  } else {
    return count + " Bet";
  }
}

console.log(cc(2));
console.log(cc(3));
