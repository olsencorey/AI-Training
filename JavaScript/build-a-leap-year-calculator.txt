function isLeapYear(num) {
  let result;

  if (num % 400 === 0) {
    result = true;
  } else if (num % 100 ===0) {
    result = false;
  } else if (num % 4 === 0) {
    result = true;
  } else {
    result = false;
  }

  if (result) {
    return `${num} is a leap year.`;
  } else {
    return `${num} is not a leap year.`;
  }
}

let year = 1000;
let result = isLeapYear(year);
console.log(result);
